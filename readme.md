- 取得現在時間
  ```javascript
  // now 2025-04-16 14:34:18

  tt    // [Item 1] [2025-04-16 14:34:18]: 1744785258
  tt @n // [Item 1] [2025-04-16 14:34:18]: 1744785258
  ```

- 用時間戳取得時間
  ```javascript

  // unix timestamp(sec)
  tt 1744785777    // [Item 1] [2025-04-16 14:42:57]: 1744785777
  // unix timestamp(ms)
  tt 1744785777124 // [Item 1] [2025-04-16 14:42:57]: 1744785777

  // unix timestamp(ms), but in base-36 format
  tt 0m9jfp9fg     // [Item 1] [2025-04-16 12:31:21]: 1744777881
  ```

- 用時間取得時間戳
  ```javascript
  // now 2025-04-16 14:34:50

  // 僅指定月份跟日期，時間用當前時間
  tt 1206,     // [Item 1] [2025-12-06 14:34:50]: 1765002890
  // 僅指定年份、月份跟日期，時間用當前時間
  tt 251206,   // [Item 1] [2025-12-06 14:34:50]: 1765002890
  // 僅指定年份、月份跟日期，時間用當前時間 (年份使用full year)
  tt 20251206  // [Item 1] [2025-12-06 14:34:50]: 1765002890

  tt ,1500     // [Item 1] [2025-04-16 15:00:00]: 1744786800
  tt 1500      // [Item 1] [2025-04-16 15:00:00]: 1744786800

  tt 1206,1644 // [Item 1] [2025-12-06 16:44:00]: 1765010640

  tt 1206,0    // [Item 1] [2025-12-06 00:00:00]: 1764950400
  ```

- 用多個時間取得多個時間戳
  ```javascript

  // 取得多個時間
  tt 1206, 1744786198 1500
  /*
    [Item 1] [2025-12-06 14:51:29]: 1765003889
    [Item 2] [2025-04-16 14:49:58]: 1744786198
    [Item 3] [2025-04-16 15:00:00]: 1744786800
  */
  ```


- 時間的增加與減少
  ```javascript

  // 往後十日
  tt @n +d10
  // [Item 1] [2025-04-16 14:49:58]: 1744786198
  // [Item 2] [2025-04-26 14:49:58]: 1745650198

  // 往後十月
  tt @n +M10
  // [Item 1] [2025-04-16 14:53:04]: 1744786384
  // [Item 2] [2026-02-16 14:53:04]: 1771224784

  // 往後十年
  tt @n +y10
  // [Item 1] [2025-04-16 14:53:44]: 1744786424
  // [Item 2] [2035-04-16 14:53:44]: 2060319224

  // 往後1年2月3日1時30分45秒
  tt 250416,0 +y1M2d3h1m30s45
  // [Item 1] [2025-04-16 00:00:00]: 1744732800
  // [Item 2] [2026-06-19 01:30:45]: 1781803845

  // 往前1.5分 (1分30秒)
  tt ,1500 -m1.5
  // [Item 1] [2025-04-16 15:00:00]: 1744786800
  // [Item 2] [2025-04-16 14:58:30]: 1744786710
  ```

- 時間相減
  ```javascript

  tt @n - 228,1300
  // 2025-04-16 14:59:09 - 2025-02-28 13:00:00
  // 1 個月  17 日  1 小時  59 分鐘  9 秒  (6 週 5 日)  (47 日)
  ```

- 設定時區
  ```javascript

  // 設定時區為+7
  tt @set 7
  tt
  // Timezone: UTC 7
  // UTC+8 [LOCAL TIME] [2025-04-16 15:00:10]
  // [Item 1] [2025-04-16 14:00:10]: 1744786810

  // 還原時區為系統時區
  tt @reset // Reset timezone: UTC 8
  tt @rs // Reset timezone: UTC 8
  ```
